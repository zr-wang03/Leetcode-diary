class Solution:
    def findMaxForm(self, strs: List[str], m: int, n: int) -> int:
        dp=(m+1)*[(n+1)*[0]]

        for string in strs:
            ones=0
            zeros=0
            for letter in string:
                if letter=="0":
                    zeros+=1
                else:
                    ones+=1
            
            for i in range(m,zeros,-1):
                print(i)
                for j in range(n,ones,-1):
                    dp[i][j]=max(dp[i][j],dp[i-zeros][j-ones]+1);
            print(dp)          
        return dp[m][n]
